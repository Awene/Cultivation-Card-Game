  // 预处理：修复 AI 生成的非标准 YAML 格式
  function fixYaml(raw){
    // 1. 去除 <char_info> 标签
    raw = raw.replace(/<\/?char_info>/g,'').trim();
    // 2. 修复 标签 行：把 [A][B][C] 这种连续方括号格式转成带引号的字符串
    //    匹配行首空白 + "标签:" + 后面内容，只要后面包含 ][ 就需要处理
    raw = raw.replace(/([ \t]*标签\s*:\s*)(\S.*)/gm, function(_, prefix, val){
      val = val.trim();
      // 已经是合法 YAML 数组 [a, b] 且不含连续 ][ 则跳过
      if(/^\[/.test(val) && !/\]\s*\[/.test(val)) return prefix + val;
      // 提取所有 [...] 内的内容
      var items = [];
      val.replace(/\[([^\]]*)\]/g, function(_m, inner){ items.push(inner.trim()); });
      if(items.length === 0){
        // 无括号格式，直接引号包裹
        return prefix + '"' + val.replace(/"/g,'\\"') + '"';
      }
      return prefix + '[' + items.map(function(s){ return '"'+s.replace(/"/g,'\\"')+'"'; }).join(', ') + ']';
    });
    return raw;
  }
